var request_example = tFormer( 'request_example', {
	fields: {
		field_name: {
			url   : 'ajax.html',
			method: 'post',
			data  : { // this params will be in post request
				custom_param: 'custom value',
				twitter     : '@TjRus'
			},
			start : function () {}, // function executed when ajax request starts
			end   : function ( result ) { // function executed when ajax request ends
				// `result` is what your request returns
				// here you should decide is the field valid or no and return true/false.
				return false;
			}
		}
	},
	submit: function ( event ) {
		if ( event.preventDefault ) {
			event.preventDefault();
		} else {
			event.returnValue = false;
		}
		// rewriting default submit code
		alert( 'Your request form verified successfully.' );
		request_example.processingOff();
		return false;
	}
} );

var date_example = tFormer( 'date_example', {
	fields: {
		date_hard: {
			rules: '*', // have to tell the field that it is required
			own: function(){ // own validation function
				return _v_(this.value).validateWithRules({rules: '*|D=Y-M-D R:I:S', rule_separator: '|'});
			}
		}
	},
	submit: function ( event ) {
		if ( event.preventDefault ) {
			event.preventDefault();
		} else {
			event.returnValue = false;
		}
		// rewriting default submit code
		alert( 'Your form verified successfully.' );
		date_example.processingOff();
		return false;
	}
} );
